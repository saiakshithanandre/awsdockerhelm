# To install the awsdocker Helm chart, use the following command:
helm install awsdocker-release . -n awsdocker

# To uninstall the awsdocker Helm chart, use the following command:
helm uninstall awsdocker-release -n awsdocker

# To Upgrade the awsdocker Helm chart, use the following command:
helm upgrade awsdocker-release . -n awsdocker
